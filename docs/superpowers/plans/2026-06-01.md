  async def generate_case_title_and_summary(
      task: dict[str, Any],
      report: dict[str, Any],
      llm: Any | None,
  ) -> tuple[str, str]:
      """调用 Kimi K2.5
  生成概括性案例标题（10-20字）和公开摘要（50-120字）。
      LLM 不可用时降级为规则生成。
      """
      from app.services.analysis_persistence import
  normalize_final_verdict

      verdict_payload =
  normalize_final_verdict(report.get("verdict_payload") or
  {})
      verdict = report.get("verdict") or
  verdict_payload.get("verdict") or "inconclusive"
      confidence = report.get("confidence_overall") or
  verdict_payload.get("confidence_overall")
      files = _task_files(task)
      media_category = _derive_media_category(files,
  task.get("input_type"))
      difficulty = _difficulty(confidence)

      if llm is None:
          return _fallback_title_and_summary(verdict,
  media_category, confidence, difficulty)

      # 构建 prompt 输入
      file_names = [
          redact_public_text(str(f.get("name") or
  "未知文件"))[:60]
          for f in files[:5]
      ]
      case_prompt =
  normalize_case_prompt(task.get("description") or
  _metadata(task).get("case_prompt") or "")[:200]
      key_evidence = verdict_payload.get("key_evidence") or
  []
      evidence_text = "\n".join(
          f"- {e if isinstance(e, str) else json.dumps(e,
  ensure_ascii=False)}"
          for e in key_evidence[:3]
      )
      verdict_label = VERDICT_LABEL_MAP.get(verdict,
  verdict)
      category_label =
  CATEGORY_LABEL_MAP.get(media_category, media_category)
      conf_text = f"{float(confidence) * 100:.1f}%" if
  confidence is not None else "未知"

      prompt = f"""你是一名数字取证专家，需要为一个公开案例
  库生成案例标题和摘要。

  案例信息：
  - 检材类型：{category_label}
  - 研判结论：{verdict_label}
  - 综合置信度：{conf_text}
  - 难度：{difficulty}
  - 检材文件：{', '.join(file_names) if file_names else 
  '未知'}
  - 案例描述：{case_prompt if case_prompt else '无'}
  - 关键证据：
  {evidence_text if evidence_text else '- 无'}

  请生成：
  1. 标题：10-20字，概括性命名，不含文件名，如"政府通知图文
  伪造案"
  2. 摘要：50-120字，面向公众，说明案件背景、检材类型、结论

  以 JSON 格式返回，只返回 JSON，不要其他内容：
  {{"title": "...", "summary": "..."}}"""

      try:
          from langchain_core.messages import HumanMessage
          response = await
  llm.ainvoke([HumanMessage(content=prompt)])
          text = response.content.strip()
          # 提取 JSON（可能被 markdown 代码块包裹）
          json_match = re.search(r'\{[^{}]+\}', text,
  re.DOTALL)
          if json_match:
              data = json.loads(json_match.group())
              title =
  redact_public_text(str(data.get("title") or
  "")).strip()[:120]
              summary =
  redact_public_text(str(data.get("summary") or
  "")).strip()[:360]
              if title and summary:
                  return title, summary
      except Exception as exc:
          logger.warning("LLM case title/summary generation
  failed, using fallback: %s", exc)

      return _fallback_title_and_summary(verdict,
  media_category, confidence, difficulty)
